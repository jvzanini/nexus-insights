   ```ts
   import { pgPool } from "@/lib/pg-pool";
   import { fromZonedTime, toZonedTime, formatInTimeZone } from "date-fns-tz";
   import { startOfDay, endOfDay, startOfMonth, endOfMonth, startOfWeek, endOfWeek } from "date-fns";

   const cache = new Map<string, { value: string; expiresAt: number }>();
   const TTL_MS = 60_000;

   async function getSetting(key: string, defaultValue: string): Promise<string> {
     const cached = cache.get(key);
     if (cached && cached.expiresAt > Date.now()) return cached.value;
     try {
       const r = await pgPool.query<{ value: unknown }>(
         "SELECT value FROM app_settings WHERE key = $1",
         [key],
       );
       const value = (r.rows[0]?.value as string) || defaultValue;
       cache.set(key, { value, expiresAt: Date.now() + TTL_MS });
       return value;
     } catch {
       return defaultValue;
     }
   }

   export async function getPlatformTz(): Promise<string> {
     return getSetting("platform.timezone", "America/Sao_Paulo");
   }

   export async function getPlatformLocale(): Promise<string> {
     return getSetting("platform.locale", "pt-BR");
   }

   export function invalidatePlatformSettings(): void {
     cache.clear();
   }

   export type PeriodKey = "hoje" | "semana_atual" | "mes_atual" | "custom";

   export function getPeriodInTz(
     key: PeriodKey,
     tz: string,
     customRange?: { start: Date; end: Date },
   ): { start: Date; end: Date } {
     const nowInTz = toZonedTime(new Date(), tz);
     switch (key) {
       case "hoje": {
         const start = fromZonedTime(startOfDay(nowInTz), tz);
         const end = fromZonedTime(endOfDay(nowInTz), tz);
         return { start, end };
       }
       case "semana_atual": {
         const start = fromZonedTime(startOfWeek(nowInTz, { weekStartsOn: 1 }), tz);
         const end = fromZonedTime(endOfWeek(nowInTz, { weekStartsOn: 1 }), tz);
         return { start, end };
       }
       case "mes_atual": {
         const start = fromZonedTime(startOfMonth(nowInTz), tz);
         const end = fromZonedTime(endOfMonth(nowInTz), tz);
         return { start, end };
       }
       case "custom": {
         if (!customRange) throw new Error("customRange obrigatório quando key=custom");
         return {
           start: fromZonedTime(startOfDay(toZonedTime(customRange.start, tz)), tz),
           end: fromZonedTime(endOfDay(toZonedTime(customRange.end, tz)), tz),
         };
       }
     }
   }

   export function formatDateInTz(d: Date | string, tz: string, locale: string, opts?: Intl.DateTimeFormatOptions): string {
     const date = typeof d === "string" ? new Date(d) : d;
     return formatInTimeZone(date, tz, opts?.dateStyle === "long" ? "dd 'de' MMMM 'de' yyyy" : "dd/MM/yyyy");
   }
   ```

4. Criar `src/lib/actions/platform-settings.ts`:
   ```ts
   "use server";
   import { pgPool } from "@/lib/pg-pool";
   import { invalidatePlatformSettings } from "@/lib/datetime";
   import { redis } from "@/lib/redis";
   import { auth } from "@/auth";

   export async function updatePlatformSettings(input: { timezone?: string; locale?: string }) {
     const session = await auth();
     if ((session?.user as { platformRole?: string })?.platformRole !== "super_admin") {
       throw new Error("Apenas super_admin pode editar configurações de plataforma");
     }
     // Validações
     if (input.timezone) {
       try { new Intl.DateTimeFormat("en-US", { timeZone: input.timezone }); }
       catch { throw new Error("Timezone IANA inválido"); }
     }
     if (input.locale && !/^[a-z]{2,3}(-[A-Z]{2,4})?$/.test(input.locale)) {
       throw new Error("Locale BCP-47 inválido");
     }
     const updates: Array<[string, string]> = [];
     if (input.timezone) updates.push(["platform.timezone", input.timezone]);
     if (input.locale) updates.push(["platform.locale", input.locale]);
     for (const [key, value] of updates) {
       await pgPool.query(
         `INSERT INTO app_settings (key, value, category, updated_at)
          VALUES ($1, to_jsonb($2::text), 'platform', NOW())
          ON CONFLICT (key) DO UPDATE SET value = to_jsonb($2::text), updated_at = NOW()`,
         [key, value],
       );
     }
     invalidatePlatformSettings();
     // Invalida cache reports (TZ mudou → ranges em cache estão errados)
     try {
       const keys = await redis.keys("report:*");
       if (keys.length > 0) await redis.del(...keys);
     } catch {}
     return { ok: true };
   }
   ```

   ```sql
   -- Atendentes mais rápidos (avg first_response_time)
   SELECT u.name, AVG(EXTRACT(EPOCH FROM cr.first_response_time)) as avg_seconds
   FROM conversations c
   JOIN reporting_events cr ON cr.conversation_id = c.id AND cr.name = 'first_response'
   JOIN users u ON u.id = c.assignee_id
   WHERE c.account_id = $1 AND c.created_at >= $2
   GROUP BY u.id, u.name

   LIMIT 5;

   JOIN users u ON u.id = c.assignee_id

   GROUP BY u.id, u.name
   ORDER BY count DESC
   LIMIT 5;

   SELECT i.name, count(c.id) as count
   FROM conversations c
   JOIN inboxes i ON i.id = c.inbox_id
   WHERE c.status = 0 AND c.account_id = $1
   GROUP BY i.id, i.name
   ORDER BY count DESC
   LIMIT 5;
   ```

   ```ts
     url.pathname = pathname;
   ```
